1. Перший API Route

В Next.js App Router API створюються через файл route.js в папці app/api/. Кожен файл route.js — це Route Handler, який обробляє HTTP-запити. На відміну від page.js, який рендерить UI, route.js повертає дані (зазвичай JSON).

Конвенція: Файл має називатись саме route.js (не api.js, не handler.js). Next.js автоматично маршрутизує запити на основі структури папок: app/api/test/route.js → /api/test.

Завдання: Створити тестовий API Route

Створіть файл app/api/test/route.js — простий ендпоінт, що повертає JSON:

app/api/test/route.js
import { NextResponse } from 'next/server'

export async function GET() {
  return NextResponse.json({
    message: 'API працює! Кав\'ярня Кофеїн',
    timestamp: new Date().toISOString()
  })
}
Зверніть увагу:
  • NextResponse.json() — зручний метод для повернення JSON-відповіді
  • Функція називається GET — це обробник HTTP GET-запитів
  • Функція async — Route Handlers завжди асинхронні
✅ Контрольна точка
  • Відкрийте http://localhost:3000/api/test в браузері
  • Має відображатись JSON: {"message": "API працює! ...", "timestamp": "..."}
  • При кожному оновленні timestamp змінюється

2. API для списку напоїв

Зараз дані напоїв зберігаються в lib/drinks.js як константний масив. Щоб API міг додавати, оновлювати та видаляти напої, потрібно зробити масив змінюваним та додати допоміжні функції.

⚠️ Важливо: In-memory дані

Дані зберігаються в пам'яті сервера. При перезапуску npm run dev всі зміни зникнуть. Це нормально для навчання — на тижні 7-8 ми підключимо MongoDB для постійного зберігання.

let nextId = 11

export { drinks }

export function getDrinkById(id) {
  return drinks.find((drink) => drink.id === Number(id))
}

export function getCategories() {
  return ["Всі", ...new Set(drinks.map((item) => item.category))]
}

export function addDrink(data) {
  const newDrink = {
    id: nextId++,
    name: data.name,
    description: data.description || '',
    price: Number(data.price),
    emoji: data.emoji || '☕',
    category: data.category || 'Інше',
    available: data.available !== undefined ? data.available : true
  }
  drinks.push(newDrink)
  return newDrink
}

export function updateDrink(id, data) {
  const index = drinks.findIndex((drink) => drink.id === Number(id))
  if (index === -1) return null

  drinks[index] = {
    ...drinks[index],
    ...data,
    id: drinks[index].id // ID не змінюється
  }
  return drinks[index]
}

export function deleteDrink(id) {
  const index = drinks.findIndex((drink) => drink.id === Number(id))
  if (index === -1) return null

  const deleted = drinks[index]
  drinks.splice(index, 1)
  return deleted
}
Що змінилось:
  • const drinks → let drinks — масив тепер змінюваний
  • let nextId = 11 — лічильник для нових ID (наступний після останнього)
  • addDrink(data) — створює новий напій та додає в масив
  • updateDrink(id, data) — оновлює напій за ID, повертає null якщо не знайдено
  • deleteDrink(id) — видаляє напій за ID, повертає видалений об'єкт або null

Завдання: Створити API Route для списку напоїв

Створіть файл app/api/drinks/route.js — ендпоінт з GET (список з фільтрацією) та POST (створення):

app/api/drinks/route.js
import { NextResponse } from 'next/server'
import { drinks, addDrink } from '@/lib/drinks'

// GET /api/drinks — список напоїв з фільтрацією
export async function GET(request) {
  const { searchParams } = new URL(request.url)
  const category = searchParams.get('category')
  const search = searchParams.get('search')

  let result = [...drinks]

  // Фільтрація за категорією
  if (category && category !== 'Всі') {
    result = result.filter(drink => drink.category === category)
  }

  // Пошук за назвою
  if (search) {
    result = result.filter(drink =>
      drink.name.toLowerCase().includes(search.toLowerCase())
    )
  }

  return NextResponse.json(result)
}

// POST /api/drinks — створення нового напою
export async function POST(request) {
  try {
    const body = await request.json()

    // Валідація обов'язкових полів
    if (!body.name || !body.category || !body.price) {
      return NextResponse.json(
        { error: 'Поля name, category та price є обов\'язковими' },
        { status: 400 }
      )
    }

    if (typeof body.price !== 'number' || body.price <= 0) {
      return NextResponse.json(
        { error: 'Ціна має бути додатнім числом' },
        { status: 400 }
      )
    }

    const newDrink = addDrink(body)
    return NextResponse.json(newDrink, { status: 201 })
  } catch (error) {
    return NextResponse.json(
      { error: 'Невалідний JSON' },
      { status: 400 }
    )
  }
}
Параметри запиту (Query Parameters): Об'єкт request — це стандартний Web API Request. Для отримання query-параметрів використовуємо new URL(request.url).searchParams. Наприклад, для URL /api/drinks?category=Кава отримаємо searchParams.get('category') → "Кава".
✅ Контрольна точка
  • /api/drinks — повертає масив з 10 напоїв
  • /api/drinks?category=Кава — повертає лише напої категорії «Кава»
  • /api/drinks?search=лат — повертає «Латте» та «Матча латте»

3. API для одного напою

Для отримання одного напою за ID використовуємо динамічний сегмент [id] — так само як і для сторінок. Файл app/api/drinks/[id]/route.js буде обробляти запити до /api/drinks/1, /api/drinks/2 тощо.

Завдання: Створити API Route для одного напою

Створіть файл app/api/drinks/[id]/route.js — GET ендпоінт з обробкою 404:

app/api/drinks/[id]/route.js
import { NextResponse } from 'next/server'
import { getDrinkById } from '@/lib/drinks'

// GET /api/drinks/[id] — один напій за ID
export async function GET(request, { params }) {
  const { id } = await params

  const drink = getDrinkById(id)

  if (!drink) {
    return NextResponse.json(
      { error: 'Напій не знайдено' },
      { status: 404 }
    )
  }

  return NextResponse.json(drink)
}
Зверніть увагу на await params: В Next.js 15 параметри маршруту (params) є асинхронними — їх потрібно «розпакувати» через await. Це стосується як page.js, так і route.js. Деструктуризація { params } відбувається з другого аргументу функції-обробника.
✅ Контрольна точка
  • /api/drinks/1 — повертає JSON з даними Еспресо
  • /api/drinks/3 — повертає JSON з даними Латте
  • /api/drinks/999 — повертає {"error": "Напій не знайдено"} зі статусом 404

5. Оновлення та видалення

Додамо PUT та DELETE обробники до файлу app/api/drinks/[id]/route.js. PUT — повне оновлення напою, DELETE — видалення.

Завдання: Додати PUT та DELETE обробники

Оновіть файл app/api/drinks/[id]/route.js — додайте обробники PUT та DELETE до вже існуючого GET:

app/api/drinks/[id]/route.js
import { NextResponse } from 'next/server'
import { getDrinkById, updateDrink, deleteDrink } from '@/lib/drinks'

// GET /api/drinks/[id] — один напій за ID
export async function GET(request, { params }) {
  const { id } = await params

  const drink = getDrinkById(id)

  if (!drink) {
    return NextResponse.json(
      { error: 'Напій не знайдено' },
      { status: 404 }
    )
  }

  return NextResponse.json(drink)
}

// PUT /api/drinks/[id] — оновлення напою
export async function PUT(request, { params }) {
  const { id } = await params

  try {
    const body = await request.json()

    // Валідація
    if (!body.name || !body.category || !body.price) {
      return NextResponse.json(
        { error: 'Поля name, category та price є обов\'язковими' },
        { status: 400 }
      )
    }

    const updated = updateDrink(id, body)

    if (!updated) {
      return NextResponse.json(
        { error: 'Напій не знайдено' },
        { status: 404 }
      )
    }

    return NextResponse.json(updated)
  } catch (error) {
    return NextResponse.json(
      { error: 'Невалідний JSON' },
      { status: 400 }
    )
  }
}

// DELETE /api/drinks/[id] — видалення напою
export async function DELETE(request, { params }) {
  const { id } = await params

  const deleted = deleteDrink(id)

  if (!deleted) {
    return NextResponse.json(
      { error: 'Напій не знайдено' },
      { status: 404 }
    )
  }

  return NextResponse.json({
    message: `Напій "${deleted.name}" видалено`,
    deleted
  })
}

6. Підключення dashboard до API

Тепер, коли API готовий, підключимо до нього dashboard. Сторінки, що отримують та змінюють дані, стануть Client Components з використанням useState, useEffect та fetch.

Чому dashboard/page.js (статистика) залишається Server Component? Сторінка статистики (app/dashboard/page.js) тільки читає дані та не потребує інтерактивності. Вона може напряму імпортувати дані з lib/drinks.js — це працює, бо Server Components виконуються на сервері. Не все потрібно робити через API — серверний код може працювати з даними напряму.

Завдання: Конвертувати список напоїв на Client Component з fetch

Оновіть app/dashboard/drinks/page.js — замість прямого імпорту даних, отримуйте їх з API через fetch:

app/dashboard/drinks/page.js
'use client'

import { useState, useEffect } from 'react'
import Link from 'next/link'
import DrinkActions from '@/components/DrinkActions'

export default function DrinksPage() {
  const [drinks, setDrinks] = useState([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState(null)

  async function fetchDrinks() {
    try {
      setLoading(true)
      const response = await fetch('/api/drinks')
      if (!response.ok) throw new Error('Помилка завантаження')
      const data = await response.json()
      setDrinks(data)
    } catch (err) {
      setError(err.message)
    } finally {
      setLoading(false)
    }
  }

  useEffect(() => {
    fetchDrinks()
  }, [])

  async function handleDelete(id) {
    if (!confirm('Видалити цей напій?')) return

    try {
      const response = await fetch(`/api/drinks/${id}`, {
        method: 'DELETE'
      })
      if (!response.ok) throw new Error('Помилка видалення')
      // Оновлюємо список після видалення
      fetchDrinks()
    } catch (err) {
      alert(err.message)
    }
  }

  if (loading) {
    return (
      <div className="flex justify-center items-center py-12">
        <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-amber-700"></div>
      </div>
    )
  }

  if (error) {
    return (
      <div className="bg-red-50 border border-red-200 rounded p-4">
        <p className="text-red-600">{error}</p>
        <button onClick={fetchDrinks} className="mt-2 text-blue-600 underline">
          Спробувати знову
        </button>
      </div>
    )
  }

  return (
    <div>
      <div className="flex justify-between items-center mb-6">
        <h1 className="text-3xl font-bold">Напої ({drinks.length})</h1>
        <Link
          href="/dashboard/drinks/new"
          className="bg-amber-700 text-white px-4 py-2 rounded hover:bg-amber-800"
        >
          + Додати напій
        </Link>
      </div>

      <div className="bg-white rounded-lg shadow overflow-hidden">
        <table className="w-full">
          <thead className="bg-gray-50">
            <tr>
              <th className="px-6 py-3 text-left text-sm font-medium text-gray-500">Назва</th>
              <th className="px-6 py-3 text-left text-sm font-medium text-gray-500">Категорія</th>
              <th className="px-6 py-3 text-left text-sm font-medium text-gray-500">Ціна</th>
              <th className="px-6 py-3 text-left text-sm font-medium text-gray-500">Наявність</th>
              <th className="px-6 py-3 text-left text-sm font-medium text-gray-500">Дії</th>
            </tr>
          </thead>
          <tbody>
            {drinks.map(drink => (
              <tr key={drink.id} className="border-t hover:bg-gray-50">
                <td className="px-6 py-4">
                  <Link href={`/dashboard/drinks/${drink.id}`} className="text-blue-600 hover:underline">
                    {drink.emoji} {drink.name}
                  </Link>
                </td>
                <td className="px-6 py-4 text-gray-600">{drink.category}</td>
                <td className="px-6 py-4 text-gray-600">{drink.price} грн</td>
                <td className="px-6 py-4">
                  <span className={`px-2 py-1 rounded text-sm ${
                    drink.available
                      ? 'bg-green-100 text-green-800'
                      : 'bg-red-100 text-red-800'
                  }`}>
                    {drink.available ? 'В наявності' : 'Немає'}
                  </span>
                </td>
                <td className="px-6 py-4">
                  <button
                    onClick={() => handleDelete(drink.id)}
                    className="text-red-600 hover:text-red-800"
                  >
                    Видалити
                  </button>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    </div>
  )
}
Ключові зміни:
  • 'use client' — сторінка тепер Client Component
  • useState для drinks, loading, error
  • useEffect з fetchDrinks() — завантаження при монтуванні
  • handleDelete — видалення через API та перезавантаження списку
  • Стани loading та error обробляються в UI

Завдання: Конвертувати сторінку деталей на Client Component

app/dashboard/drinks/[id]/page.js
'use client'

import { useState, useEffect, use } from 'react'
import Link from 'next/link'

export default function DrinkDetailPage({ params }) {
  const { id } = use(params)
  const [drink, setDrink] = useState(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState(null)

  useEffect(() => {
    async function fetchDrink() {
      try {
        const response = await fetch(`/api/drinks/${id}`)

          if (response.status === 404) {

          }

        setDrink(data)
      } catch (err) {
        setError(err.message)
      } finally {
        setLoading(false)
      }
    }

    fetchDrink()
  }, [id])

  if (loading) {

        <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-amber-700"></div>
      </div>
    )
  }

  if (error) {
    return (
      <div className="text-center py-12">
        <h1 className="text-4xl font-bold text-gray-400 mb-4">404</h1>
        <p className="text-gray-600 mb-4">{error}</p>
        <Link href="/dashboard/drinks" className="text-blue-600 hover:underline">
          &larr; До списку напоїв
        </Link>
      </div>
    )
  }

  return (
    <div>
      <Link href="/dashboard/drinks" className="text-blue-600 hover:underline">
        &larr; Назад до списку
      </Link>

      <div className="bg-white rounded-lg shadow p-8 mt-4">
        <div className="flex justify-between items-start mb-6">
          <div className="flex items-center gap-4">
            <span className="text-4xl">{drink.emoji}</span>
            <h1 className="text-3xl font-bold">{drink.name}</h1>
          </div>

            drink.available
              ? 'bg-green-100 text-green-800'
              : 'bg-red-100 text-red-800'
          }`}>
            {drink.available ? 'В наявності' : 'Немає'}
          </span>
        </div>

        <div className="grid grid-cols-2 gap-6 mb-6">
          <div>
            <p className="text-sm text-gray-500">Категорія</p>
            <p className="text-lg font-medium">{drink.category}</p>

            <p className="text-gray-700">{drink.description}</p>
          </div>
        )}

    </div>
  )

Зверніть увагу на use(params): В Client Component не можна використовувати await для розпакування params (бо компонент не async). Замість цього використовуємо React хук use(), який «розгортає» Promise синхронно під час рендеру.

Завдання: Оновити форму створення напою

Оновіть app/dashboard/drinks/new/page.js — відправляйте дані через POST запит до API:

app/dashboard/drinks/new/page.js
'use client'

import { useState } from 'react'
import { useRouter } from 'next/navigation'
import Link from 'next/link'

export default function NewDrinkPage() {
  const router = useRouter()
  const [saving, setSaving] = useState(false)
  const [error, setError] = useState(null)

  async function handleSubmit(e) {
    e.preventDefault()
    setSaving(true)
    setError(null)

    const formData = new FormData(e.target)
    const data = {
      name: formData.get('name'),
      category: formData.get('category'),
      price: Number(formData.get('price')),
      description: formData.get('description'),
      emoji: formData.get('emoji') || '☕',
      available: formData.get('available') === 'on'
    }

    try {
      const response = await fetch('/api/drinks', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(data)
      })

      if (!response.ok) {
        const result = await response.json()
        throw new Error(result.error || 'Помилка створення')
      }

      router.push('/dashboard/drinks')
    } catch (err) {
      setError(err.message)
    } finally {
      setSaving(false)
    }
  }

  return (
    <div>
      <Link href="/dashboard/drinks" className="text-blue-600 hover:underline">
        &larr; Назад до списку
      </Link>

      <h1 className="text-3xl font-bold mt-4 mb-6">Додати напій</h1>

      {error && (
        <div className="bg-red-50 border border-red-200 rounded p-4 mb-4">
          <p className="text-red-600">{error}</p>
        </div>
      )}

      <form onSubmit={handleSubmit} className="bg-white rounded-lg shadow p-6 max-w-lg">
        <div className="mb-4">
          <label className="block text-sm font-medium text-gray-700 mb-1">Назва *</label>
          <input
            type="text"
            name="name"
            required
            className="w-full border rounded px-3 py-2 focus:ring-2 focus:ring-amber-500 focus:border-transparent"
          />
        </div>

        <div className="mb-4">
          <label className="block text-sm font-medium text-gray-700 mb-1">Категорія *</label>
          <select
            name="category"
            required
            className="w-full border rounded px-3 py-2 focus:ring-2 focus:ring-amber-500 focus:border-transparent"
          >
            <option value="Кава">Кава</option>
            <option value="Чай">Чай</option>
            <option value="Їжа">Їжа</option>
            <option value="Інше">Інше</option>
          </select>
        </div>

        <div className="mb-4">
          <label className="block text-sm font-medium text-gray-700 mb-1">Ціна (грн) *</label>
          <input
            type="number"
            name="price"
            min="1"
            required
            className="w-full border rounded px-3 py-2 focus:ring-2 focus:ring-amber-500 focus:border-transparent"
          />
        </div>

        <div className="mb-4">
          <label className="block text-sm font-medium text-gray-700 mb-1">Emoji</label>
          <input
            type="text"
            name="emoji"
            placeholder="☕"
            className="w-full border rounded px-3 py-2 focus:ring-2 focus:ring-amber-500 focus:border-transparent"
          />
        </div>

        <div className="mb-4">
          <label className="block text-sm font-medium text-gray-700 mb-1">Опис</label>
          <textarea
            name="description"
            rows="3"
            className="w-full border rounded px-3 py-2 focus:ring-2 focus:ring-amber-500 focus:border-transparent"
          ></textarea>
        </div>

        <div className="mb-6">
          <label className="flex items-center gap-2">
            <input type="checkbox" name="available" defaultChecked />
            <span className="text-sm text-gray-700">В наявності</span>
          </label>
        </div>

        <button
          type="submit"
          disabled={saving}
          className="bg-amber-700 text-white px-6 py-2 rounded hover:bg-amber-800 disabled:opacity-50"
        >
          {saving ? 'Збереження...' : 'Створити напій'}
        </button>
      </form>
    </div>
  )
}
Ключові моменти форми:
  • e.preventDefault() — запобігаємо стандартній відправці форми
  • FormData — збираємо дані з усіх полів форми
  • fetch('/api/drinks', { method: 'POST', ... }) — відправляємо JSON на API
  • router.push('/dashboard/drinks') — перенаправлення після успішного створення
  • Стан saving блокує кнопку під час відправки

Завдання: Оновити компонент DrinkActions

Якщо у вас є окремий компонент components/DrinkActions.jsx з кнопкою видалення, оновіть його для роботи через API:

components/DrinkActions.jsx
'use client'

import { useRouter } from 'next/navigation'

export default function DrinkActions({ drinkId, drinkName }) {
  const router = useRouter()

  async function handleDelete() {
    if (!confirm(`Видалити "${drinkName}"?`)) return

    try {
      const response = await fetch(`/api/drinks/${drinkId}`, {
        method: 'DELETE'
      })

      if (!response.ok) {
        throw new Error('Помилка видалення')
      }

      router.push('/dashboard/drinks')
      router.refresh()
    } catch (err) {
      alert(err.message)
    }
  }

  return (
    <div className="flex gap-2">
      <button
        onClick={handleDelete}
        className="bg-red-600 text-white px-4 py-2 rounded hover:bg-red-700"
      >
        Видалити
      </button>
    </div>
  )
}
Зверніть увагу: router.refresh() після router.push() примусово оновлює кеш сторінки. Без нього Next.js може показати застарілі дані з кешу навігації.
✅ Контрольна точка
  • /dashboard/drinks — показує список напоїв, завантажених з API
  • /dashboard/drinks/1 — показує деталі напою з API
  • /dashboard/drinks/999 — показує повідомлення «Напій не знайдено»
  • /dashboard/drinks/new — форма створює напій через POST, перенаправляє до списку
  • Кнопка «Видалити» в таблиці та на сторінці деталей працює через DELETE API
  • Loading-спіннер показується під час завантаження даних

Підсумок

Критерії успішного виконання

  • GET /api/test повертає JSON з message та timestamp
  • GET /api/drinks повертає список напоїв
  • Фільтрація за категорією (?category=) працює
  • Пошук (?search=) працює
  • GET /api/drinks/[id] повертає один напій
  • Неіснуючий ID повертає 404 з повідомленням
  • POST /api/drinks створює напій з валідацією
  • PUT /api/drinks/[id] оновлює напій
  • DELETE /api/drinks/[id] видаляє напій
  • Dashboard /dashboard/drinks отримує дані з API
  • Сторінка деталей /dashboard/drinks/[id] працює через API
  • Форма створення /dashboard/drinks/new відправляє POST
  • Кнопка видалення виконує DELETE через API
  • Код адаптовано під тему свого варіанту

Бонусні завдання

Ці завдання не є обов'язковими і не впливають на наступні практичні роботи.

  • Реалізувати PATCH ендпоінт для часткового оновлення (без обов'язкових полів)
  • Додати сортування через query-параметри (?sort=price&order=asc)
  • Реалізувати пагінацію (?page=1&limit=5) з meta-інформацією (total, pages)
  • Створити API для другої сутності вашого проекту (наприклад, категорії, замовлення тощо)
